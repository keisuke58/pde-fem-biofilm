  % bonded / crack line
  \draw[very thick, orange!70!black] (0,1.15) -- (2.3,1.15);
  \draw[very thick, orange!70!black, dashed] (2.3,1.15) -- (4.4,1.15);
  \node[dl, orange!70!black, anchor=south] at (2.3,1.18) {crack tip};
  % Mode I : opening (normal) — arrows at x=1.1
  \draw[crackar, blue!60!black] (1.1,1.45) -- (1.1,2.15);
  \draw[crackar, blue!60!black] (1.1,0.85) -- (1.1,0.15);
  \node[dl, blue!60!black, anchor=east, align=right] at (-0.2,1.15)
    {Mode\,I\\ (opening)\\ $t_n,\,\delta_n$};
  % Mode II : sliding (tangential) — arrows at x=3.3
  \draw[crackar, red!60!black] (3.0,1.75) -- (3.75,1.75);
  \draw[crackar, red!60!black] (3.4,0.55) -- (2.65,0.55);

  \node[dl, anchor=north, align=center] at (2.2,-0.35)
    {effective opening $\delta_m=\sqrt{\delta_n^2+\delta_t^2}$,\;

  \node[font=\footnotesize\bfseries] at (2.2,-1.55) {(a) mode I / II decomposition};

  % small traction-separation inset (defines G = area), centred above
  \begin{scope}[shift={(1.15,2.8)}]

    \fill[orange!16] (0,0) -- (0.6,1.2) -- (1.8,0) -- cycle;
    \draw[blue!55!black, thick] (0,0) -- (0.6,1.2) -- (1.8,0);

    \node[dl, anchor=west] at (0.62,1.2) {$t^{0}$};

% ══ (b) Benzeggagh–Kenane critical-energy envelope ═════════════════
\begin{scope}[shift={(9.0,0)}]
  \draw[ax] (0,0) -- (5.6,0) node[right, dl]{mode mixity $B=G_{II}/G_T$};
  \draw[ax] (0,0) -- (0,4.0) node[above, dl]{critical $G_c$};
  % tick labels
  \node[dl, anchor=north] at (0,0) {$0$ (pure I)};
  \node[dl, anchor=north] at (5.0,0) {$1$ (pure II)};
  % B-K curve: G_c = G_Ic + (G_IIc - G_Ic) * B^eta  (G_Ic=1.6, G_IIc=3.2, eta=1.8)
  \draw[blue!55!black, very thick, samples=80, domain=0:5, smooth]
     plot (\x, {1.6 + 1.6*(\x/5)^1.8});
  % endpoints
  \fill[red!70!black] (0,1.6) circle (2pt);
  \fill[red!70!black] (5.0,3.2) circle (2pt);
  \draw[dashed, gray!55] (0,1.6) -- (-0.05,1.6) node[left, dl]{$G_{Ic}$};
  \draw[dashed, gray!55] (5.0,3.2) -- (5.0,0);
  \draw[dashed, gray!55] (5.0,3.2) -- (0,3.2) node[left, dl]{$G_{IIc}$};
  \node[dl, blue!55!black, anchor=west] at (1.7,2.05)
    {$G_c=G_{Ic}+(G_{IIc}-G_{Ic})\,B^{\eta}$};
  \node[dl, anchor=west] at (2.4,1.45) {B–K exponent $\eta$};
  \node[font=\footnotesize\bfseries] at (2.6,-1.35) {(b) Benzeggagh–Kenane $G_c(B)$};
\end{scope}

% ── BC callouts ─────────────────────────────────────────────────────
\node[bc, right=1.0cm of fluid] (topbc) {%
  \textbf{Top surface (biofilm)}\\

  \textbf{Bottom (implant base)}\\

\node[bc, left=1.0cm of bio] (latbc) {%
  \textbf{Lateral (sides)}\\

% arrows from callouts to edges
\draw[ar] (topbc.west) -- (fluid.east);

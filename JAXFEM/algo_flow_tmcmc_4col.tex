\node[Nm, below=0.45cm of kg] (nh) {%

\node[Nm, below=0.45cm of nh] (ct) {%

  $\phantom{{}=}{+}\tfrac{\mu{-}\lambda\ln J}{J}
    (\delta_{ik}\delta_{jl}{+}\delta_{il}\delta_{jk})$};

%% COLUMN A  (x = 0)   Phase 0: INPUT  +  Phase 1: PDE SYSTEM
%% ══════════════════════════════════════════════════════════════════════
\node[Ng] (H) {%
  Hamilton functional (Junker 2021):\\[2pt]
  $\mathcal{H}=\!\int_\tau\!(G+C-B+D)\,\mathrm{d}t$\\
  $\delta\mathcal{H}=0 \;\Rightarrow$ Eq.\,34--36};

\node[Ng, below=0.5cm of H] (par) {%
  Table\,2 parameters:\\[2pt]
  $\mu{=}3.356$,\; $E{=}10$,\; $\nu{=}0.49$\\
  $\beta{=}2$,\; $k_\alpha{=}10^{-3}$,\;
  $r{=}100$,\; $g{=}10^8$};

\node[Nb, below=0.8cm of par] (e34) {%
  \textbf{Eq.\,34} — biofilm $\phi\in[0,1]$:\\[2pt]
  $\dot\phi = \beta\Delta\phi - k_\alpha\alpha
    + \dfrac{rc}{k+c}(\nabla\phi\cdot\hat{n}_c)$};

\node[Nb, below=0.5cm of e34] (e35) {%
  \textbf{Eq.\,35} — nutrient $c$:\\[2pt]
  $\dot{c} = D\Delta c - g\,\phi$};

\node[Nb, below=0.5cm of e35] (e36) {%
  \textbf{Eq.\,36} — growth var.\ $\alpha$:\\[2pt]
  $\dot\alpha = k_\alpha\phi$\quad
  (Gauss-pt local ODE)};

%% ══════════════════════════════════════════════════════════════════════
%% COLUMN B  (x = 6.2cm)   Phase 2: FD SOLVER  +  Phase 3: GROWTH MECH
%% ══════════════════════════════════════════════════════════════════════
\node[Nc, right=1.1cm of H] (ini) {%
  Init: $\phi^0$,\;$c^0$,\;$\alpha^0{=}0$\\
  $60{\times}60$ grid, $20\,\mu\mathrm{m}^2$};

\node[Nc, below=0.5cm of ini] (sc) {%
  Solve $c$ (quasi-static):\\[2pt]
  $-D\Delta c + g\phi = 0$};

\node[Nc, below=0.5cm of sc] (up) {%
  Update $\phi$ (explicit Euler, $\Delta t{=}8{\times}10^{-4}$):\\[2pt]
  $\phi \mathrel{+}= \Delta t\!\left[
    \beta\nabla^2\phi - k_\alpha\alpha
    + \tfrac{rc}{k+c}\nabla\phi{\cdot}\hat{n}_c
  \right]$};

\node[Nc, below=0.5cm of up] (ua) {%
  $\alpha \mathrel{+}= \Delta t\,k_\alpha\,\phi$};

\node[dec, below=0.5cm of ua] (tc) {$t{+}{=}\Delta t$\\ $t < T_{\max}$?};

\node[Nc, below=0.6cm of tc] (sn) {%
  Snapshot $\alpha(x,y)$ at\\
  4 time points};

%% ── Phase 3 (below snapshot, same column) ────────────────────────────
\node[Nm, below=1.2cm of sn] (kg) {%
  Growth kinematics:\\[2pt]
  $\mathbf{F}_g{=}\mathrm{diag}(1,1,\lambda_z)$\quad
  $\lambda_z{=}1{+}\max(0,\beta_d\phi_{Pg}{+}b)$\\
  $\mathbf{F}_e{=}\mathbf{F}\mathbf{F}_g^{-1}$,\quad
  $J{=}\det\mathbf{F}_e$};

\node[Nm, below=0.5cm of kg] (nh) {%
  neo-Hookean Cauchy stress:\\[2pt]
  $\boldsymbol\sigma
    = \tfrac{\mu}{J}(\mathbf{b}_e{-}\mathbf{I})
    + \tfrac{\lambda}{J}\ln J\;\mathbf{I}$\\
  $\mathbf{b}_e = \mathbf{F}_e\mathbf{F}_e^\top$};

\node[Nm, below=0.5cm of nh] (ct) {%
  Consistent spatial tangent (B\&W §6.5):\\[2pt]
  $\mathbb{C}_{ijkl}
    = \tfrac{\lambda}{J}\delta_{ij}\delta_{kl}
    + \tfrac{\mu{-}\lambda\ln J}{J}
      (\delta_{ik}\delta_{jl}{+}\delta_{il}\delta_{jk})$};

%% ══════════════════════════════════════════════════════════════════════

\node[Nv, below=0.5cm of inp] (gp) {%

\node[Nv, below=0.5cm of gp] (um) {%
  UMAT \texttt{umat\_klempt\_alpha.f}\\[2pt]
  In: $\mathbf{F}$, $\phi_{Pg}$, PROPS, STATEV\\
  Out: STRESS $\boldsymbol\sigma$, DDSDDE $\mathbb{C}$\\
  SDV: $\lambda_z$,\;$J_e$,\;$\alpha$};

\node[Nv, below=0.5cm of um] (as) {%
  FEM assembly:\\[2pt]

\node[dec, below=0.5cm of as] (nr) {$\|\mathbf{R}\|/\|\mathbf{R}_0\|$\\$<\varepsilon$?};

\node[Nv, right=0.9cm of nr] (it) {%
  $\delta\mathbf{u}{\leftarrow}\mathbf{K}_T^{-1}\mathbf{R}$\\

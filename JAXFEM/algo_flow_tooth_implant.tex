\node[Ng, below=2.2cm of td] (H) {%
  Hamilton principle (Junker 2021):\\[2pt]

  $\delta\mathcal{H}{=}0$};

\node[Ng, below=0.5cm of H] (par2) {%

\node[Nk, below=1.8cm of par2] (e34) {%
  \textbf{Eq.\,34} — biofilm $\phi_{total}$:\\[2pt]

    {+}\tfrac{rc}{k+c}(\nabla\phi{\cdot}\hat{n}_c)$};

\node[Nk, below=0.5cm of e34] (e35) {%
  \textbf{Eq.\,35} — nutrient $c$:\\[2pt]
  $\dot c = D\Delta c - g\,\phi_{total}$};

\node[Nk, below=0.5cm of e35] (e36) {%
  \textbf{Eq.\,36} — growth $\alpha$:\\[2pt]
  $\dot\alpha = k_\alpha\phi_{total}$\quad (local ODE)};

\node[Nm, right=0.8cm of tooth] (alpmap) {%

  interpolated to Gauss points};

\node[Nm, below=0.5cm of alpmap] (Fsplit) {%
  Growth kinematics\; $\mathbf{F}{=}\mathbf{F}_e\mathbf{F}_g$:\\[2pt]
  $\lambda_z{=}1{+}\max(0,\beta_d\phi_{Pg}{+}b)$\\
  $\mathbf{F}_e{=}\mathbf{F}\mathbf{F}_g^{-1}$,\; $J{=}\det\mathbf{F}_e$};

\node[Nm, below=0.5cm of Fsplit] (nhS) {%

  $\boldsymbol\sigma{=}\tfrac{\mu}{J}(\mathbf{b}_e{-}\mathbf{I})
    {+}\tfrac{\lambda}{J}\ln J\,\mathbf{I}$};

%%── Ph.4 FEM NR ───────────────────────────────────────────────────
\node[Nv, below=1.8cm of nhS] (umatcall) {%

  Out: $(\boldsymbol\sigma,\mathbb{C})$};

\node[Nv, below=0.5cm of umatcall] (KT) {%
  FEM assembly + Newton-Raphson:\\[2pt]
  $\mathbf{K}_T\delta\mathbf{u}{=}{-}\mathbf{R}$,\;
  until $\|\mathbf{R}\|/\|\mathbf{R}_0\|{<}\varepsilon$};
